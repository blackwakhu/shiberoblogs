How to change background image on hover in react and svelte using tailwindcss 

Derrick Shibero Wakhu May 28, 2025

React

React js is a very popular front-end library. 

import images from './assets/images.jpg'
import red from './assets/red.jpg'

function App() {
  return (
    <div>
      <div className="m-2 p-2 flex-col border-1">
        <h1 className="font-bold text-center">Changing the background image on hover in react</h1>
        <div className='flex justify-center relative items-center m-10 w-100 h-100'>
          <div className="group w-100 h-100 overflow-hidden rounded shadow">
            <img src={red} alt="images" className='absolute inset-0 w-full h-full object-cover transition-opacity duration-300 group-hover:opacity-0' />
            <img src={images} alt="red" className='absolute inset-0 w-full h-full object-cover opacity-0 transition-opacity duration-300 group-hover:opacity-100' />
          </div>
        </div>
      </div>
    </div>
  )
}
